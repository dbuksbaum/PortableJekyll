def safe_hash_access(hsh, keys)
  key = keys.shift
  return nil unless hsh.is_a?(Hash) && hsh.has_key?(key)

  if keys.empty?
    hsh[key]
  else
    safe_hash_access hsh[key], keys
  end
end
