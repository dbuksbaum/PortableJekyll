=> :bar

def find_in_hash_list(collection, key)
  result = collection.find { |hsh| hsh.keys.include?(key) }
  result ? result[key] : nil
end

def guard_hash(hsh, keys)
  key = keys.shift
  return nil unless hsh.is_a?(Hash) && hsh.has_key?(key)

  if keys.empty?
    hsh[key]
  else
    guard_hash hsh[key], keys
  end
end
