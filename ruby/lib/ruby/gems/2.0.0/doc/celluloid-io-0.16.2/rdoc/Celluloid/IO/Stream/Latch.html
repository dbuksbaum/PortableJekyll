def synchronize
  actor = Thread.current[:celluloid_actor]
  return yield unless actor

  if @owner || @waiters > 0
    @waiters += 1
    @condition.wait
    @waiters -= 1
  end

  @owner = Task.current

  begin
    ret = yield
  ensure
    @owner = nil
    @condition.signal if @waiters > 0
  end

  ret
end
