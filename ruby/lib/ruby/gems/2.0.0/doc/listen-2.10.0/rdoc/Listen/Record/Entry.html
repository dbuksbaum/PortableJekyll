def children
  child_relative = _join
  (Dir.entries(sys_path) - %w(. ..)).map do |name|
    Entry.new(@root, child_relative, name)
  end
end

def _join
  args = [@relative, @name].compact
  args.empty? ? nil : ::File.join(*args)
end
