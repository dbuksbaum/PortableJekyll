def inspect
  attrs = [nil]
  members.each do |mem|
    if value = send(mem)
      attrs << "@#{mem}=#{value.inspect}"
    end
  end
  if !custom_members.empty?
    attrs << "@custom=#{custom_members.inspect}"
  end
  %Q(#<#{self.class}#{attrs.join(" ")}>)
end
