def generate(site)
  site.pages.each { |page| emojify page }
  site.posts.each { |post| emojify post }
  site.docs_to_write.each { |doc| emojify doc }

def src
  @src ||=
    if config.key?("emoji") && config["emoji"].key?("src")
      config["emoji"]["src"]
    else
      "https://assets.github.com/images/icons/"
    end
end
