def _deep_find_all(key, object = self, matches = [])
  deep_locate_result = Hashie::Extensions::DeepLocate.deep_locate(key, object).tap do |result|
    result.map! { |element| element[key] }
  end

  matches.concat(deep_locate_result)
end
