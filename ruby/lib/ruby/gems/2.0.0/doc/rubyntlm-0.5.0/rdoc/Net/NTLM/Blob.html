def parse(str, offset=0)
  # 28 is the length of all fields before the variable-length
  # target_info field.
  if str.size > 28
    enable(:target_info)
    # Grab everything except the last 4 bytes (which will be :unknown2)
    self[:target_info].value = str[28..-5]
  end
  super
end
