def self.hex_inspect(str)
  str.codepoints.map { |c| c.to_s(16).rjust(4, '0') }.join('-')

def image_filename
  if custom?
    '%s.png' % name
  else
    'unicode/%s.png' % hex_inspect.sub(/-fe0f\b/, '')
  end
end
