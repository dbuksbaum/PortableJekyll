    store_computed_value(key, new_value) click to toggle source 
    # File lib/thread_safe/non_concurrent_cache_backend.rb, line 126
def store_computed_value(key, new_value)
  if new_value.nil?
    @backend.delete(key)
    nil
  else
    @backend[key] = new_value
  end
end
